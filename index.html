async function parseOrders() {

  const raw = document.getElementById('order-input').value.trim();

  if (!raw) { setStatus('error','Please paste some orders first.'); return; }


  const key = document.getElementById('api-key-input').value.trim();

  if (!key) {

    setStatus('error', 'Please enter your OpenAI API key above.');

    return;

  }


  setStatus('info',

    '<span class="spinner"></span> Parsing orders with AI…');


  const systemPrompt = `You are an order parsing assistant for a CBD 

artisan product company. Extract every order from the raw text and 

return a JSON array. Each order object must have these exact fields:

{

  "date": "DD.M.YY or DD.MM.YY string exactly as written, 

           carry forward the last seen date if no new date appears",

  "customer": "full customer name",

  "address": "full address on one line, comma separated",

  "mz": "MZ tracking number or empty string",

  "gs": number (quantity in grams for GS product, 0 if none),

  "gp": number (quantity in grams for GP product, 0 if none),

  "gk": number (quantity in grams for GK product, 0 if none),

  "xtc": number (quantity in grams for XTC product, 0 if none),

  "notes": "any special instructions or flags, empty string if none"

}

Rules:

- Product codes: GS, GP, GK, XTC. 

  Sum multiple quantities for same product on one order

  (e.g. "7GP + 1GP" = 8 GP).

- Ignore MZ tracking numbers when calculating products.

- If an order line has a special note (e.g. RS, NS, BOX, DECOY, 

  separate parcels) put it in notes.

- Dates carry forward until a new date appears in the text.

- Return ONLY valid JSON array, no markdown, no explanation.`;


  try {

    const res = await fetch('https://api.openai.com/v1/chat/completions', {

      method: 'POST',

      headers: {

        'Content-Type': 'application/json',

        'Authorization': `Bearer ${key}`

      },

      body: JSON.stringify({

        model: 'gpt-4o-mini',

        messages: [

          { role: 'system', content: systemPrompt },

          { role: 'user',   content: raw }

        ],

        temperature: 0

      })

    });


    if (!res.ok) {

      const err = await res.json();

      setStatus('error',

        `API error: ${err.error?.message || res.status}`);

      return;

    }


    const data    = await res.json();

    const content = data.choices[0].message.content.trim();

    let parsed;


    try {

      // Strip markdown code fences if present

      const clean = content

        .replace(/^```json\s*/,'').replace(/```$/,'').trim();

      parsed = JSON.parse(clean);

    } catch (e) {

      setStatus('error',

        'Could not parse AI response. Check the console for details.');

      console.error(content);

      return;

    }


    if (!Array.isArray(parsed) || parsed.length === 0) {

      setStatus('error', 'No orders found in that text.');

      return;

    }


    // Add IDs and merge

    parsed.forEach(o => {

      o.id = Date.now() + Math.random();

      o.gs  = parseFloat(o.gs)  || 0;

      o.gp  = parseFloat(o.gp)  || 0;

      o.gk  = parseFloat(o.gk)  || 0;

      o.xtc = parseFloat(o.xtc) || 0;

    });


    orders = orders.concat(parsed);

    saveStorage();


    setStatus('success',

      `✅ ${parsed.length} orders imported successfully.`);

    showPreview(parsed);

    document.getElementById('order-input').value = '';

    refreshAll();


  } catch (e) {

    setStatus('error', `Network error: ${e.message}`);

  }

}

function renderOrdersTable() {

  const search  = (document.getElementById('search-input')?.value||'')

                  .toLowerCase();

  const dateF   = document.getElementById('filter-date')?.value  || '';

  const prodF   = document.getElementById('filter-product')?.value || '';


  // Populate date filter

  const dateSelect = document.getElementById('filter-date');

  const dates = [...new Set(orders.map(o => o.date))].sort(dateSort);

  const curDate = dateSelect.value;

  dateSelect.innerHTML = '<option value="">All Dates</option>' +

    dates.map(d =>

      `<option value="${d}" ${d===curDate?'selected':''}>${d}</option>`

    ).join('');


  let filtered = orders.filter(o => {

    const matchSearch = !search ||

      (o.customer||'').toLowerCase().includes(search) ||

      (o.address ||'').toLowerCase().includes(search);

    const matchDate   = !dateF || o.date === dateF;

    const matchProd   = !prodF || o[prodF.toLowerCase()] > 0;

    return matchSearch && matchDate && matchProd;

  });


  // Sort

  filtered.sort((a, b) => {

    let av = a[sortKey], bv = b[sortKey];

    if (sortKey === 'date') {

      av = parseDateVal(a.date);

      bv = parseDateVal(b.date);

    }

    if (av < bv) return sortAsc ? -1 :  1;

    if (av > bv) return sortAsc ?  1 : -1;

    return 0;

  });


  const tbody = document.getElementById('orders-body');

  const tfoot = document.getElementById('orders-foot');


  tbody.innerHTML = filtered.map(o => `

    <tr>

      <td style="white-space:nowrap">${o.date}</td>

      <td>${esc(o.customer)}</td>

      <td style="font-size:0.8rem;color:var(--muted)">

        ${esc(o.address)}

      </td>

      <td style="font-size:0.78rem;color:var(--muted)">

        ${esc(o.mz)}

      </td>

      <td>${o.gs  > 0 ?

        `<span class="qty-badge badge-gs">${o.gs}g</span>`

        :'<span class="badge-zero">—</span>'}</td>

      <td>${o.gp  > 0 ?

        `<span class="qty-badge badge-gp">${o.gp}g</span>`

        :'<span class="badge-zero">—</span>'}</td>

      <td>${o.gk  > 0 ?

        `<span class="qty-badge badge-gk">${o.gk}g</span>`

        :'<span class="badge-zero">—</span>'}</td>

      <td>${o.xtc > 0 ?

        `<span class="qty-badge badge-xtc">${o.xtc}g</span>`

        :'<span class="badge-zero">—</span>'}</td>

      <td class="notes-cell">${esc(o.notes)}</td>

      <td>

        <button class="btn btn-danger btn-small"

                onclick="deleteOrder(${o.id})">✕</button>

      </td>

    </tr>`).join('');


  // Footer totals

  const t = calcTotals(filtered);

  tfoot.innerHTML = `<tr>

    <td colspan="4">

      ${filtered.length} orders shown

    </td>

    <td class="color-gs">${fmt(t.gs)}g</td>

    <td class="color-gp">${fmt(t.gp)}g</td>

    <td class="color-gk">${fmt(t.gk)}g</td>

    <td class="color-xtc">${fmt(t.xtc)}g</td>

    <td colspan="2"></td>

  </tr>`;

}

    ? (((retail-cost)/retail)*100).toFixed(1) : 0;

  const el = document.getElementById(`result-${p}`);

  if (el) el.innerHTML = `

    Profit per gram: <strong>£${profit}</strong><br>

    Margin: <strong>${margin}%</strong>`;


  if (document.getElementById('panel-profit')

      ?.classList.contains('active')) {

    renderProfit();

  }

}


function renderProfit() {

  // Rebuild price cards with current labels

  buildPriceCards();


  const products = ['GS','GP','GK','XTC'];

  const totals   = calcTotals(orders);

  const tbody    = document.getElementById('profit-body');

  const tfoot    = document.getElementById('profit-foot');


  let grandCost=0, grandRev=0, grandProfit=0;


  tbody.innerHTML = products.map(p => {

    const qty    = totals[p.toLowerCase()];

    const c      = prices[p].cost;

    const r      = prices[p].retail;

    const cost   = qty * c;

    const rev    = qty * r;

    const profit = rev - cost;

    const margin = rev > 0

      ? ((profit/rev)*100).toFixed(1) : '—';

    grandCost   += cost;

    grandRev    += rev;

    grandProfit += profit;

    return `<tr>

      <td><strong>${labels[p]}</strong>

          <span style="color:var(--muted);font-size:0.8rem">

          (${p})</span></td>

      <td>${fmt(qty)}g</td>

      <td>£${fmt2(c)}</td>

      <td>£${fmt2(r)}</td>

      <td>£${fmt2(cost)}</td>

      <td>£${fmt2(rev)}</td>

      <td class="${profit>=0?'profit-positive':'profit-negative'}">

        £${fmt2(profit)}

      </td>

      <td class="${profit>=0?'profit-positive':'profit-negative'}">

        ${margin}%

      </td>

    </tr>`;

  }).join('');


  const gm = grandRev > 0

    ? (((grandProfit)/grandRev)*100).toFixed(1) : '—';

  tfoot.innerHTML = `<tr>

    <td colspan="4">TOTAL</td>

    <td>£${fmt2(grandCost)}</td>

    <td>£${fmt2(grandRev)}</td>

    <td class="${grandProfit>=0?'profit-positive':'profit-negative'}">

      £${fmt2(grandProfit)}

    </td>

    <td class="${grandProfit>=0?'profit-positive':'profit-negative'}">

      ${gm}%

    </td>

  </tr>`;


  // Profit by date

  const byDate = groupByDate(orders);

  const dates  = Object.keys(byDate).sort(dateSort);

  const pdBody = document.getElementById('profit-date-body');

  const pdFoot = document.getElementById('profit-date-foot');


  let tGs=0,tGp=0,tGk=0,tXtc=0,tTot=0;


  pdBody.innerHTML = dates.map(d => {

    const rows = byDate[d];

    const t    = calcTotals(rows);

    const pGs  = t.gs  * (prices.GS.retail  - prices.GS.cost);

    const pGp  = t.gp  * (prices.GP.retail  - prices.GP.cost);

    const pGk  = t.gk  * (prices.GK.retail  - prices.GK.cost);

    const pXtc = t.xtc * (prices.XTC.retail - prices.XTC.cost);

    const tot  = pGs + pGp + pGk + pXtc;

    tGs+=pGs; tGp+=pGp; tGk+=pGk; tXtc+=pXtc; tTot+=tot;

    const pc = v =>

      `<td class="${v>=0?'profit-positive':'profit-negative'}">

        £${fmt2(v)}

      </td>`;

    return `<tr>

      <td>${d}</td>

      ${pc(pGs)}${pc(pGp)}${pc(pGk)}${pc(pXtc)}${pc(tot)}

    </tr>`;

  }).join('');


  const pc2 = v =>

    `<td class="${v>=0?'profit-positive':'profit-negative'}">

      £${fmt2(v)}

    </td>`;

  pdFoot.innerHTML = `<tr>

    <td>TOTAL</td>

    ${pc2(tGs)}${pc2(tGp)}${pc2(tGk)}${pc2(tXtc)}${pc2(tTot)}

  </tr>`;

}

function exportCSV() {

  const headers = [

    'Date','Customer','Address','MZ Number',

    'GS (g)','GP (g)','GK (g)','XTC (g)',

    'GS Revenue','GP Revenue','GK Revenue','XTC Revenue',

    'Total Revenue','Gross Profit','Notes'

  ];


  const rows = orders.map(o => {

    const gsRev  = o.gs  * prices.GS.retail;

    const gpRev  = o.gp  * prices.GP.retail;

    const gkRev  = o.gk  * prices.GK.retail;

    const xtcRev = o.xtc * prices.XTC.retail;

    const totRev = gsRev + gpRev + gkRev + xtcRev;

    const profit =

      o.gs  * (prices.GS.retail  - prices.GS.cost)  +

      o.gp  * (prices.GP.retail  - prices.GP.cost)  +

      o.gk  * (prices.GK.retail  - prices.GK.cost)  +

      o.xtc * (prices.XTC.retail - prices.XTC.cost);

    return [

      o.date, o.customer, o.address, o.mz,

      o.gs, o.gp, o.gk, o.xtc,

      gsRev.toFixed(2), gpRev.toFixed(2),

      gkRev.toFixed(2), xtcRev.toFixed(2),

      totRev.toFixed(2), profit.toFixed(2),

      o.notes

    ].map(v => `"${String(v||'').replace(/"/g,'""')}"`);

  });


  const csv = [headers, ...rows].map(r => r.join(',')).join('\n');

  const blob = new Blob([csv], { type: 'text/csv' });

  const url  = URL.createObjectURL(blob);

  const a    = document.createElement('a');

  a.href     = url;

  a.download = `cbd-orders-${

    new Date().toISOString().slice(0,10)}.csv`;

  a.click();

  URL.revokeObjectURL(url);

}

function calcTotals(rows) {

  return rows.reduce((acc, o) => {

    acc.gs  += o.gs  || 0;

    acc.gp  += o.gp  || 0;

    acc.gk  += o.gk  || 0;

    acc.xtc += o.xtc || 0;

    return acc;

  }, { gs:0, gp:0, gk:0, xtc:0 });

}


function calcRevenue(totals) {

  return totals.gs  * prices.GS.retail  +

         totals.gp  * prices.GP.retail  +

         totals.gk  * prices.GK.retail  +

         totals.xtc * prices.XTC.retail;

}


function calcProfitTotal(totals) {

  return totals.gs  * (prices.GS.retail  - prices.GS.cost)  +

         totals.gp  * (prices.GP.retail  - prices.GP.cost)  +

         totals.gk  * (prices.GK.retail  - prices.GK.cost)  +

         totals.xtc * (prices.XTC.retail - prices.XTC.cost);

}


function groupByDate(rows) {

  return rows.reduce((acc, o) => {

    const d = o.date || 'Unknown';

    if (!acc[d]) acc[d] = [];

    acc[d].push(o);

    return acc;

  }, {});

}


function parseDateVal(s) {

  if (!s) return 0;

  const parts = s.split('.');

  if (parts.length < 3) return 0;

  const [d,m,y] = parts;

  const year = parseInt(y) + (parseInt(y) < 100 ? 2000 : 0);

  return new Date(year, parseInt(m)-1, parseInt(d)).getTime();

}


function dateSort(a, b) {

  return parseDateVal(a) - parseDateVal(b);

}


function esc(s) {

  return String(s||'')

    .replace(/&/g,'&amp;')

    .replace(/</g,'&lt;')

    .replace(/>/g,'&gt;');

}


function fmt(n)  { return parseFloat((n||0).toFixed(2)); }

function fmt2(n) { return parseFloat((n||0).toFixed(2)).toFixed(2); }
